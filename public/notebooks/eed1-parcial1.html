import numpy as np

periods = [
    (2000, 2023),
    (2000, 2003),
    (2003, 2007),
    (2007, 2015),
    (2015, 2019),
    (2019, 2023)
]

results = {}

for start_year, end_year in periods:
    # Filter data for the current period
    df_period = df[(df['año de titulación'] >= start_year) & (df['año de titulación'] <= end_year)]

    # Get the 'femenino' data for the period
    femenino_data = df_period['femenino']

    # Calculate mean, median, and mode
    mean_femenino = femenino_data.mean()
    median_femenino = femenino_data.median()

    # Calculate mode, handling cases where there might be multiple modes or no mode
    mode_femenino = femenino_data.mode()
    if not mode_femenino.empty:
        mode_femenino_val = mode_femenino.tolist() # Store all modes as a list
    else:
        mode_femenino_val = "No mode"

    results[f'{start_year}-{end_year}'] = {
        'mean': mean_femenino,
        'median': median_femenino,
        'mode': mode_femenino_val
    }

# Print the results
for period, stats in results.items():
    print(f"Período: {period}")
    print(f"  Media del personal femenino: {stats['mean']:.2f}")
    print(f"  Mediana del personal femenino: {stats['median']:.2f}")
    print(f"  Moda del personal femenino: {stats['mode']}")
    print("\n")

# Mujeres a elegir para la comisión

# Calcular combinaciones para mujeres
combinaciones_mujeres = comb(num_mujeres_elegibles, mujeres_comision)
