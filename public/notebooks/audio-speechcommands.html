1. Carga del dataset

torchaudio.datasets.SPEECHCOMMANDS descarga el dataset y devuelve, por cada item, una tupla (waveform, sample_rate, label, speaker_id, utterance_number). Nos quedamos con 5 comandos para que la demo sea rápida y balanceada.

# 🧠 Prompt: "Definí una subclase de SPEECHCOMMANDS que filtre solo los comandos elegidos y un subset chico."
COMANDOS = ["yes", "no", "up", "down", "stop"]   # 5 clases (subir para versión completa)
MAX_POR_CLASE = 300                               # demo; subir/quitar en GPU completa

class SubsetSC(torchaudio.datasets.SPEECHCOMMANDS):
    def __init__(self, subset, comandos, max_por_clase):
        super().__init__(root="./", download=True, subset=subset)
        # filtramos los walker paths que correspondan a los comandos elegidos
        keep, contador = [], {c: 0 for c in comandos}
        for p in self._walker:
            label = os.path.normpath(p).split(os.sep)[-2]
            if label in contador and contador[label] < max_por_clase:
                keep.append(p); contador[label] += 1
        self._walker = keep

train_set = SubsetSC("training",   COMANDOS, MAX_POR_CLASE)
test_set  = SubsetSC("testing",    COMANDOS, max(60, MAX_POR_CLASE // 4))
print("Train clips:", len(train_set), "| Test clips:", len(test_set))

wav, sr, label, *_ = train_set[0]
print("Ejemplo -> shape:", wav.shape, "| sample_rate:", sr, "| label:", label)
LABELS = sorted(COMANDOS)
label2idx = {l: i for i, l in enumerate(LABELS)}
print("Mapeo de clases:", label2idx)

2. Del audio al espectrograma de Mel

Un MelSpectrogram transforma la onda 1D en una matriz 2D (frecuencias-Mel × tiempo), que es lo que "ve" la CNN. Visualizamos un ejemplo.

# 🧠 Prompt: "Definí el transform MelSpectrogram a 8kHz y graficá onda + espectrograma de un clip."
SAMPLE_RATE = 8000   # resampleamos de 16k a 8k para aligerar
N_MELS = 64

resample = torchaudio.transforms.Resample(orig_freq=16000, new_freq=SAMPLE_RATE)
mel = torchaudio.transforms.MelSpectrogram(sample_rate=SAMPLE_RATE, n_fft=512,
                                           hop_length=128, n_mels=N_MELS)
to_db = torchaudio.transforms.AmplitudeToDB()

wav, sr, label, *_ = train_set[0]
wav_rs = resample(wav)
spec = to_db(mel(wav_rs))

fig, ax = plt.subplots(1, 2, figsize=(14, 4))
ax[0].plot(wav_rs.t().numpy()); ax[0].set_title(f"Forma de onda — '{label}'")
ax[0].set_xlabel("muestras")
im = ax[1].imshow(spec.squeeze().numpy(), origin="lower", aspect="auto", cmap="magma")
ax[1].set_title("Espectrograma de Mel (dB)"); ax[1].set_xlabel("tiempo"); ax[1].set_ylabel("Mel")
plt.colorbar(im, ax=ax[1]); plt.tight_layout(); plt.show()

3. Dataset con padding a tamaño fijo

Los clips no duran exactamente lo mismo → los espectrogramas tienen distinto largo temporal. Para apilarlos en batches los llevamos a un ancho fijo (FIXED_FRAMES) con padding (ceros) o recorte. Envolvemos todo en un Dataset que devuelve (spec, label_idx).

# 🧠 Prompt: "Creá un Dataset wrapper que resamplee, calcule el MelSpectrograma en dB y haga pad/crop a frames fijos."
FIXED_FRAMES = 64   # ancho temporal fijo del espectrograma

class AudioMelDataset(Dataset):
    def __init__(self, sc_dataset):
        self.ds = sc_dataset
    def __len__(self):
        return len(self.ds)
    def __getitem__(self, i):
        wav, sr, label, *_ = self.ds[i]
        wav = resample(wav)
        spec = to_db(mel(wav))                 # [1, n_mels, T]
        T = spec.shape[-1]
        if T < FIXED_FRAMES:
            spec = F.pad(spec, (0, FIXED_FRAMES - T))
        else:
            spec = spec[..., :FIXED_FRAMES]
        return spec, label2idx[label]

train_ds = AudioMelDataset(train_set)
test_ds  = AudioMelDataset(test_set)

train_loader = DataLoader(train_ds, batch_size=64, shuffle=True,  num_workers=2)
test_loader  = DataLoader(test_ds,  batch_size=64, shuffle=False, num_workers=2)

xb, yb = next(iter(train_loader))
print("Batch de espectrogramas:", xb.shape, "(esperado [B,1,64,64])  | labels:", yb.shape)

4. Modelo: CNN sobre espectrogramas

Una CNN compacta (2 bloques Conv→ReLU→MaxPool + clasificador denso). Trata el espectrograma como una imagen de 1 canal.

# 🧠 Prompt: "Definí una CNN con dos bloques convolucionales y un clasificador denso adaptable al número de clases."
class AudioCNN(nn.Module):
    def __init__(self, n_classes):
        super().__init__()
        self.features = nn.Sequential(
            nn.Conv2d(1, 16, 3, padding=1), nn.ReLU(), nn.BatchNorm2d(16), nn.MaxPool2d(2),
            nn.Conv2d(16, 32, 3, padding=1), nn.ReLU(), nn.BatchNorm2d(32), nn.MaxPool2d(2),
        )
        self.pool = nn.AdaptiveAvgPool2d((4, 4))   # robusto al tamaño exacto de entrada
        self.classifier = nn.Sequential(
            nn.Flatten(), nn.Dropout(0.3),
            nn.Linear(32 * 4 * 4, 64), nn.ReLU(),
            nn.Linear(64, n_classes),
        )
    def forward(self, x):
        return self.classifier(self.pool(self.features(x)))

model = AudioCNN(len(LABELS)).to(device)
out = model(xb.to(device))
print("Salida de prueba:", out.shape, f"(esperado [B, {len(LABELS)}])")

5. Entrenamiento

# 🧠 Prompt: "Definí evaluate y train que registren loss y accuracy de train y test por época."
@torch.no_grad()
def evaluate(model, loader, criterion):
    model.eval()
    loss_sum, correct, n = 0.0, 0, 0
    for xb, yb in loader:
        xb, yb = xb.to(device), yb.to(device)
        out = model(xb)
        loss_sum += criterion(out, yb).item() * xb.size(0)
        correct += (out.argmax(1) == yb).sum().item(); n += xb.size(0)
    return loss_sum / n, correct / n

def train(model, train_loader, test_loader, epochs=8, lr=1e-3):
    criterion = nn.CrossEntropyLoss()
    optimizer = torch.optim.Adam(model.parameters(), lr=lr)
    hist = {"tr_loss": [], "te_loss": [], "tr_acc": [], "te_acc": []}
    for ep in range(epochs):
        model.train()
        for xb, yb in train_loader:
            xb, yb = xb.to(device), yb.to(device)
            optimizer.zero_grad()
            loss = criterion(model(xb), yb); loss.backward(); optimizer.step()
        trl, tra = evaluate(model, train_loader, criterion)
        tel, tea = evaluate(model, test_loader, criterion)
        for k, v in zip(hist, [trl, tel, tra, tea]): hist[k].append(v)
        print(f"Época {ep+1}/{epochs} | train loss {trl:.3f} acc {tra:.3f} | test loss {tel:.3f} acc {tea:.3f}")
    return hist

EPOCHS = 8   # subir en GPU completa
hist = train(model, train_loader, test_loader, epochs=EPOCHS)
# 🧠 Prompt: "Graficá las curvas de loss y accuracy de train y test."
fig, ax = plt.subplots(1, 2, figsize=(14, 5))
ax[0].plot(hist["tr_loss"], "o-", label="train"); ax[0].plot(hist["te_loss"], "s-", label="test")
ax[0].set_title("Loss"); ax[0].set_xlabel("época"); ax[0].legend()
ax[1].plot(hist["tr_acc"], "o-", label="train"); ax[1].plot(hist["te_acc"], "s-", label="test")
ax[1].set_title("Accuracy"); ax[1].set_xlabel("época"); ax[1].legend()
plt.tight_layout(); plt.show()

6. Evaluación final

# 🧠 Prompt: "Generá predicciones en test y mostrá classification_report y matriz de confusión."
model.eval()
y_true, y_pred = [], []
with torch.no_grad():
    for xb, yb in test_loader:
        out = model(xb.to(device))
        y_pred.extend(out.argmax(1).cpu().tolist()); y_true.extend(yb.tolist())

print(classification_report(y_true, y_pred, target_names=LABELS))
cm = confusion_matrix(y_true, y_pred)
plt.figure(figsize=(7, 6))
sns.heatmap(cm, annot=True, fmt="d", cmap="Greens", xticklabels=LABELS, yticklabels=LABELS)
plt.title("Matriz de confusión — comandos de voz"); plt.ylabel("real"); plt.xlabel("predicho")
plt.tight_layout(); plt.show()

Lectura de resultados. Con un subset chico y pocas épocas la CNN ya distingue los comandos con buena accuracy. Las confusiones suelen darse entre comandos fonéticamente parecidos o cortos (p. ej. *up* vs *stop* por la oclusiva inicial), mientras que *yes/no* se separan claramente. 7. Conclusiones

  • Qué construí: un clasificador de comandos de voz con el pipeline onda → resample → MelSpectrogram (dB) → padding a tamaño fijo → CNN, con curvas de aprendizaje y matriz de confusión.
  • Aprendizajes: tratar el audio como imagen (espectrograma) permite reusar toda la maquinaria de CNN; el AmplitudeToDB ayuda a que la escala de energía sea más amigable para la red; AdaptiveAvgPool2d hace al modelo robusto al tamaño exacto del espectrograma.
  • Dificultad superada: los clips de distinta duración rompían el DataLoader al apilar el batch. Lo resolví fijando FIXED_FRAMES con F.pad/recorte en el Dataset.
  • Decisión de diseño: resampleé a 8 kHz y limité clases/clips para que corra rápido en Colab; en GPU completa subiría sample rate, clases y épocas.
  • Duda que queda: cuánto ganaría usar transfer learning de audio (p. ej. un modelo preentrenado tipo wav2vec/AST) frente a esta CNN entrenada desde cero.
